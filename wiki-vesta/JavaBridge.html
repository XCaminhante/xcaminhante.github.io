The function "generateFileListOfSources" in the SWIG bridge (KenSchalk: I assume this refers to the Java bridge which the SWIG build uses, as that's where I see that function name) should be replaced with these two fns: 

        listOfSources( fbinding: NamedFiles, prefix: text ): text
        {
                res = "";
                foreach [ src_nm = src_file ] in fbinding do
                {
                        res += if( _is_binding( src_file ) )
                                then
                                        listOfSources( src_file, prefix + src_nm )
                                else
                                        prefix + src_nm + "\n";
                };
                return res;
        };

        /**nocache**/
        generateFileListOfSources( /**pk*/ fbinding: NamedFiles ): NamedFile
        /*
                Creates a new file which contains the bound files as a newline-separated
                list
        */
        {
                res = "";
                res = listOfSources( fbinding, "" );
                return [ "files" = res ];
        };
